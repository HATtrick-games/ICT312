146  can only be one PathNode for a given state.

174  // correct from stale values

176  int numAdjacent; // -1 is unknown & needs to be queried

181 
182  bool inOpen;

184 
185  void Unlink() {
186  next->prev = prev;
187  prev->next = next;
188  next = prev = 0;
189  }
190  void AddBefore( PathNode* addThis ) {
191  addThis->next = this;
192  addThis->prev = prev;
193  prev->next = addThis;
194  prev = addThis;
195  }
196  #ifdef DEBUG
197  void CheckList()
198  {
199  MPASSERT( totalCost == FLT_MAX );
200  for( PathNode* it = next; it != this; it=it->next ) {
201  MPASSERT( it->prev == this || it->totalCost >= it->prev->totalCost );
202  MPASSERT( it->totalCost <= it->next->totalCost );
203  }
204  }
205  #endif

207  void CalcTotalCost() {
208  if ( costFromStart < FLT_MAX && estToGoal < FLT_MAX )
209  totalCost = costFromStart + estToGoal;

211  totalCost = FLT_MAX;
